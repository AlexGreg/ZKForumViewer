jimmyshiau
1 Dec 2010 06:35:55 GMT
1 Dec 2010 06:35:55 GMT

You can try

zk.Widget.$('$progressmeter').setValue(100);

It will work with ZK 5.0.6,
with ZK 505, you have to override the _fixImgWidth function
<script type="text/javascript"><![CDATA[
		zk.afterLoad('zul.wgt', function () {
			var _xProgressmeter = {}
			zk.override(zul.wgt.Progressmeter.prototype, _xProgressmeter, {
				_fixImgWidth: function () {
					var n = this.$n(), 
					img = this.$n("img");
					if (img) {
						if (zk.ie6_) img.style.width = ""; //Bug 1899749
						jq(img).animate({
							width:  Math.round((n.clientWidth * this._value) / 100) + "px"
						}, "slow");
					}		
				}
			});
		});
	]]></script>
